 public class RUNDEN {
    //schaden anfang
    private int sa;
    //schaden ende
    private int se;
    //verteidigung
    private int ve;
    
    public RUNDEN (){
      se=0;
      sa=0;
      ve=0; 
    }
    
    public int runden(int s, int v){
    sa = s;
    ve = v;
      if(ve>= 20){
       ve =19;
      }
    double z = 0;
    z = s - sa*ve/20;
    z = Math.round(z);
    se = (int)z ;
    return se;
   
    }
}
